[docs]def nodePropertyValues(g,annotkey):
    """Return a list of unique values corresponding to an existing node property:

    :param g: a graph
    :param annotkey: an existing propety name
    :return:  a list of node property values
    :rtype: list 
 
    """ 
  
    vset=set()
    pkprop= g.vertex_properties[annotkey]
    nid=0
    for v in   g.vertices():
       vset.add(pkprop[nid])  
       nid+=1
    
    pv=list(vset)
    return pv

    :param g: a graph

    :return:  a list of edge property values
    :rtype: list

[docs]def uniProtAnnotImpl( prot,conf=None):

  """Configure a mart for Uniprot to GO annotation "":

  :param prot:  list of Uniprot gene symbols
  :param conf: the configuration dictionary
  :return:  a dictionary of annotations 
  :rtype: dict 
 
  """    

  if conf is None:
    conf={
       'server':'http://www.ensembl.org',
       'mart':'ENSEMBL_MART_ENSEMBL',
       'dataset':'hsapiens_gene_ensembl',
       'attr':[
           'ensembl_gene_id',
           'external_gene_name',
           'uniprot_gn_symbol',
           'go_id'
          ],
       'searchkey' :'uniprot_gn_symbol'
     }
  
  mart=defineBiomartServer(conf['server'],conf['mart'])  
  dataset = (mart.datasets[conf['dataset']])
  attr=conf['attr']

  dataset.filters
 
  dataset._filters[conf['searchkey']]=Filter(conf['searchkey'], 'text')

  filt={conf['searchkey']: prot}

  unimap=dict()
  for p in prot:
      unimap[p] =set()
    
  res=dataset.query(attributes=attr, filters=filt)
  i=0
  for index, row in res.iterrows() : 
    i=i+1
    maxIt=1000000
    if i<= maxIt:

        uni=str(row['UniProtKB Gene Name symbol'])
        goterm=str(row['GO term accession'])
          
        if uni != "nan":

           if uni in unimap:
              ma=unimap[uni]
           else:
              ma=set()
           if goterm != "nan":     
              ma.add(goterm)  
           unimap[uni]=ma
        
    else:
        break
   
  return unimap




[docs]def uniProtAnnot(protL,conf=None,lsize=50):

   """Configure a mart for Uniprot to GO annotation "":

   :param prot:  list of Uniprot gene symbols
   :param conf: the configuration dictionary
   :param lsize: the size of each chunk of inputs to be submitted in one time
   :return:  a dictionary of annotations 
   :rtype: dict 
  
   """    
    
   chunks = [protL[x:x+lsize] for x in range(0, len(protL), lsize)]
   allM=dict()
   for prot in chunks:
     #print(prot)   
     unimap=uniProtAnnotImpl(prot)
     allM.update(unimap) 
   return allM




[docs]def annotNodeFromFile(g,annotfile,mapKey,newprop,delimiter=','):

  """Populate the nodes with a new property. The values of the property are extract from a tabular file:

  :param g: a graph
  :param annotfile: the tabular annotation file
  :param mapKey: the node property holding the primary key that must be present as a named column in the file
  :param newprop: the new property to be created that must be present as a named column in the file
  :param delimiter: tabular file delimiter
  :rtype: void 
  
  """  

  df = pd.read_csv(annotfile,   delimiter=delimiter)

  andict=dict(df.values)
  

  g.vp[newprop]=g.new_vertex_property("string")
  protL=propertyValues(g,mapKey)

  nodes=g.vertices()
  for node in nodes:
    key=g.vp[mapKey][node]
    if key is not None:
      if key in andict.keys():   
         g.vp[newprop][node]=andict[key]

[docs]def copyNodeProperties(g, sourceNode, targetNode):
    """Copy all properties of a source node to a target node:

    :param g: a graph
    :param sourceNode: source node
    :param targetNode:  target node
    :rtype: void 
  
    """ 
    for prop in g.vertex_properties.keys():
        g.vp[prop][targetNode]=g.vp[prop][sourceNode]
